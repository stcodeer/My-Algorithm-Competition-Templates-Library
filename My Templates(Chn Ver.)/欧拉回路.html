欧拉回路

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 2e5+1;

int n,m,head[N];
int tp,d[N],fr[N],to[N],ans[N];
bool vis[N];
struct nd{
	int ne,to,id;
}e[N*4];

void in(int x,int y,int w){
    static int cnt=1;
	e[++cnt].to=y;e[cnt].ne=head[x];e[cnt].id=w;head[x]=cnt;
}

void dfs(int x,int f){
	for(int &i=head[x];i;i=e[i].ne){
		if(!vis[i>>1]){
			vis[i>>1]=1;
			int now=e[i].id;
			dfs(e[i].to,x);
			ans[++ans[0]]=now;
		}
	}
	return;
}
void dfs1(int x){
	for(int &i=head[x];i;i=e[i].ne)
	if(!vis[i]){
			vis[i]=1;
			int now=e[i].id;
			dfs1(e[i].to);
			ans[++ans[0]]=now;
	}
}

int main(){
	scanf("%d",&tp); 
	if(tp==1){//Undirected Graph
		scanf("%d%d",&n,&m);
		for(int i=1,x,y;i<=m;++i){
			scanf("%d%d",&x,&y);
			in(x,y,i);in(y,x,-i);
			d[x]++;d[y]++;
		}
		for(int i=1;i<=n;++i)if(d[i]&1)return 0*printf("NO");
		for(int i=1;i<=n;++i)if(head[i]){
			dfs(i,-1);
			break;
		}
		if(ans[0]!=m)return 0*printf("NO");
		printf("YES\n");
		for(int i=ans[0];i>=1;--i)printf("%d ",ans[i]);
	}
	else{//directed Graph
		scanf("%d%d",&n,&m);
		int tag=0;
		for(int i=1,x,y;i<=m;++i){
			scanf("%d%d",&x,&y);
			in(x,y,i);
			fr[x]++;to[y]++;
		}
		for(int i=1;i<=n;++i)if(fr[i]!=to[i])return 0*printf("NO");
		for(int i=1;i<=n;++i)
        if(head[i]){
			dfs1(i);
			break;
		}
		if(ans[0]!=m)return 0*printf("NO");
		printf("YES\n");
		for(int i=ans[0];i>=1;--i)printf("%d ",ans[i]);
	}
}

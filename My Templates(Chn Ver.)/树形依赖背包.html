树形依赖背包

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 2e2+5;
const int M = 2e2+5;

int n,m,fa[N],w[N];
int totw,st[N],ed[N],p[N];
int f[N][M];
vector<int>g[N];

void dfs(int x){
	st[x]=++totw;
	p[totw]=x;
	for(int i=0;i<g[x].size();++i)dfs(g[x][i]);
	ed[x]=totw;
}

int main(){
	while(scanf("%d%d",&n,&m)!=EOF){
		if(!n&&!m)return 0;
		for(int i=0;i<=n;++i)g[i].clear();
		for(int i=1;i<=n;++i){
			scanf("%d%d",&fa[i],&w[i]);
			g[fa[i]].push_back(i);
		}
		totw=-1;dfs(0);
		for(int j=0;j<=m;++j)f[n+1][j]=0;
		for(int i=n;i>=1;--i){
			int x=p[i];
			for(int j=0;j<=m;++j)
			f[i][j]=max(j?f[st[x]+1][j-1]+w[x]:0,f[ed[x]+1][j]);
		}
		printf("%d\n",f[1][m]);
	}
}

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 64;

ll n,m,k;
struct Basis{
	ll a[N],b[N];
	vector<ll> s;

	void insert(ll x){
		for(int i=N-1;~i;--i)
		if(x&1ll<<i)x^=a[i];
		for(int i=N-1;~i;--i)
		if((x&1ll<<i)&&!a[i]){
			a[i]=x;
			for(int j=i+1;j<N;++j)
			if(a[j]&1ll<<i)a[j]^=a[i];
			return;
		}
	}
	
	void prepare(){
		for(int i=0;i<64;++i)
		if(a[i]){
			s.push_back(a[i]);
		}
	}
	
	ll kth(ll k){
		if(s.size()!=n)k--;
		ll ret=0;
		if(k>=(1ll<<s.size()))return -1;
		for(ll i=0;i<N;++i)
		if(k&(1ll<<i))ret^=s[i];
		return ret;
	}
}B;

int main(){
	cin>>n;
	ll x;

		cin>>x;
		B.insert(x);
	}
	B.prepare();

	for(int i=1;i<=m;++i){ 
		cin>>k;
		cout<<B.kth(k)<<endl;
	} 
}

三元环,四元环计数

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 1e5+5;
const ll mod = 1e9+7;

int n,m,id[N],pos[N];
vector<int> g[N],g1[N];

bool cmp(int x,int y){
	return g[x].size()<g[y].size();
} 

ll cycle3(){
	static bool vis[N];
	ll ans=0;
	for(int x=1;x<=n;++x){
		for(int y:g1[x])vis[y]=1;
		for(int y:g1[x])
		for(int z:g1[y])
		if(vis[z]){
			ans++;
		}
		for(int y:g1[x])vis[y]=0;
	}
	return ans;
}

ll cycle4(){
	static int cnt[N];
	ll ans=0;
	for(int x=1;x<=n;++x){
		for(int y:g[x])
		for(int z:g1[y])
		if(pos[z]>pos[x]){
			ans=(ans+(cnt[z]++))%mod;
		}
		for(int y:g[x])
		for(int z:g1[y])
		if(pos[z]>pos[x]){
			cnt[z]=0;
		}
	}
	return ans;
}

int main(){
	scanf("%d%d",&n,&m);
	for(int i=1,x,y;i<=m;++i){
		scanf("%d%d",&x,&y);
		g[x].push_back(y);
		g[y].push_back(x);
	}
	for(int i=1;i<=n;++i)id[i]=i;
	sort(id+1,id+n+1,cmp);
	for(int i=1;i<=n;++i)pos[id[i]]=i;
	for(int x=1;x<=n;++x)
	for(int y:g[x])
	if(pos[x]<pos[y]){
		g1[x].push_back(y);
	}
	cout<<cycle3()<<endl;
}

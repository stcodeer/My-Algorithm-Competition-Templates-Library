高斯消元

#include<bits/stdc++.h>
using namespace std;
const int N = 1e3+5;

int n;
double a[N][N],f[N];

bool gauss(){
    for(int i=1;i<=n;++i){
        int l=i;
        for(int j=i+1;j<=n;++j)if(fabs(a[j][i])>fabs(a[l][i]))l=j;
        if(fabs(a[l][i])<1e-10)return false;
        for(int j=1;j<=n+1;++j)swap(a[i][j],a[l][j]);
        double p=a[i][i];
        for(int j=1;j<=n+1;++j)a[i][j]/=p;
        for(int j=1;j<=n;++j)
        if(i!=j){
            p=a[j][i];
            for(int k=1;k<=n+1;++k)a[j][k]-=p*a[i][k];
        }
    }
    for(int i=1;i<=n;++i)f[i]=a[i][n+1];
	return true; 
}

int main(){
	cin>>n;
	for(int i=1;i<=n;++i)
	for(int j=1;j<=n+1;++j){
		cin>>a[i][j];
	}
	if(!gauss()){
		printf("No Solution\n");
	}
	else{
		for(int i=1;i<=n;++i)printf("%.2lf\n",f[i]);
	}
	return 0;
}

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int T;
ll a,b,m;

ll ph(ll x){
    ll res=x,a=x;
    for(ll i=2;i*i<=x;i++){
        if(a%i==0){
            res=res/i*(i-1);
            while(a%i==0)a/=i;
        }
    }
    if(a>1)res=res/a*(a-1);
    return res;
}

ll qpow(ll a,ll b,ll mod){
	ll r=1;
	while(b){
		if(b&1)r=r*a%mod;
		b>>=1;a=a*a%mod;
	}
	return r;
}

ll f(ll p,ll b){
    if(p==1)return 0;
    if(!b)return 1;
    ll k=ph(p);
    if(__gcd(a,p)==1)return qpow(a,f(k,b-1),p);
	long double t=0;
	for(int i=0;i<b;++i){
		t=pow(a,t);
		if(t>=k)return qpow(a,f(k,b-1)+k,p);
	}
	return qpow(a,t,p);
}

int main(){
    int T;
    scanf("%d",&T);
    while(T--){//a^a^a^a...%m(b nums in total)
        scanf("%lld%lld%lld",&a,&b,&m);
		if(!b)printf("%lld\n",1%m);
        else printf("%lld\n",f(m,b)%m);
	}
    return 0;
}

最小圆覆盖

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef double db;
const int N = 1e5+1;
 
int n,id[N];
db x[N],y[N],R;
 
db dis(int i,int j){
    return sqrt((x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]));
}
 
bool incircle(int i){
    return dis(i,0)<=R;
}
 
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;++i)scanf("%lf%lf",&x[i],&y[i]);
    for(int i=1;i<=n;++i)id[i]=i;
    random_shuffle(id+1,id+n+1);
    for(int i=1;i<=n;++i)
    if(!incircle(i)){
        x[0]=x[i];y[0]=y[i];R=0;
        for(int j=1;j<i;++j)
        if(!incircle(j)){
            x[0]=(x[i]+x[j])/2;y[0]=(y[i]+y[j])/2;R=dis(i,0);
            for(int k=1;k<j;++k)
            if(!incircle(k)){
                db a1=x[j]-x[i],b1=y[j]-y[i],c1=(a1*a1+b1*b1)/2;
                db a2=x[k]-x[i],b2=y[k]-y[i],c2=(a2*a2+b2*b2)/2;
                db d=a1*b2-a2*b1;
                x[0]=x[i]+(c1*b2-c2*b1)/d;
                y[0]=y[i]+(a1*c2-a2*c1)/d;
                R=dis(i,0);
            }
        }
    }
    printf("%.3lf\n",R);
}
